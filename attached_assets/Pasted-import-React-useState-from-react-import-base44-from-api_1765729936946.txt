import React, { useState } from 'react';
import { base44 } from '@/api/base44Client';
import { useMutation } from '@tanstack/react-query';
import { motion, AnimatePresence } from 'framer-motion';
import { 
  Heart, Package, GraduationCap, Factory, 
  CreditCard, Building2, Upload, Check, 
  ArrowRight, ArrowLeft, Loader2, Shield,
  Globe, RefreshCw, User, Mail, MapPin,
  MessageSquare, Eye, EyeOff
} from 'lucide-react';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { Textarea } from '@/components/ui/textarea';
import { RadioGroup, RadioGroupItem } from '@/components/ui/radio-group';
import { Checkbox } from '@/components/ui/checkbox';
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from '@/components/ui/select';

const donationCategories = [
  { id: 'pad', icon: Package, title: 'Donate a Pad', amount: 1, description: 'Provide sanitary pads', impact: '= 10 pads' },
  { id: 'empower', icon: Heart, title: 'Empower a Girl', amount: 2.5, description: 'Monthly support package', impact: '= 1 month support' },
  { id: 'school', icon: GraduationCap, title: 'Send Back to School', amount: 5, description: 'School supplies & support', impact: '= Full school kit' },
  { id: 'training', icon: Factory, title: 'Fund Training', amount: 50, description: 'Train a woman', impact: '= Business starter' },
];

const currencies = [
  { code: 'USD', symbol: '$', name: 'US Dollar', rate: 1 },
  { code: 'GBP', symbol: '£', name: 'British Pound', rate: 0.79 },
  { code: 'EUR', symbol: '€', name: 'Euro', rate: 0.92 },
  { code: 'NGN', symbol: '₦', name: 'Nigerian Naira', rate: 1550 },
];

const countries = [
  { code: 'US', name: 'United States' },
  { code: 'GB', name: 'United Kingdom' },
  { code: 'CA', name: 'Canada' },
  { code: 'NG', name: 'Nigeria' },
  { code: 'GH', name: 'Ghana' },
  { code: 'KE', name: 'Kenya' },
  { code: 'ZA', name: 'South Africa' },
  { code: 'DE', name: 'Germany' },
  { code: 'FR', name: 'France' },
  { code: 'AU', name: 'Australia' },
  { code: 'IN', name: 'India' },
  { code: 'AE', name: 'United Arab Emirates' },
];

export default function Donate() {
  const urlParams = new URLSearchParams(window.location.search);
  const preselectedAmount = urlParams.get('amount');
  const preselectedCategory = urlParams.get('category');
  const donationType = urlParams.get('type') || 'one_time';

  const [step, setStep] = useState(1);
  const [selectedCategory, setSelectedCategory] = useState(preselectedCategory || 'empower');
  const [amount, setAmount] = useState(preselectedAmount || '');
  const [customAmount, setCustomAmount] = useState('');
  const [currency, setCurrency] = useState('USD');
  const [isRecurring, setIsRecurring] = useState(donationType === 'monthly');
  const [paymentMethod, setPaymentMethod] = useState('bank_transfer');
  const [isAnonymous, setIsAnonymous] = useState(false);
  const [proofFile, setProofFile] = useState(null);
  const [isUploading, setIsUploading] = useState(false);
  
  const [formData, setFormData] = useState({
    donor_name: '',
    donor_email: '',
    donor_country: '',
    donor_country_code: '',
    message: '',
  });

  const [donationComplete, setDonationComplete] = useState(false);
  const [createdDonation, setCreatedDonation] = useState(null);

  const selectedCurrency = currencies.find(c => c.code === currency);
  const selectedCategoryData = donationCategories.find(c => c.id === selectedCategory);
  
  const getFinalAmount = () => {
    if (customAmount) return parseFloat(customAmount);
    if (amount) return parseFloat(amount);
    return selectedCategoryData?.amount || 0;
  };

  const getAmountInUSD = () => {
    const finalAmount = getFinalAmount();
    if (currency === 'USD') return finalAmount;
    return finalAmount / selectedCurrency.rate;
  };

  const createDonationMutation = useMutation({
    mutationFn: async (donationData) => {
      return await base44.entities.Donation.create(donationData);
    },
    onSuccess: (data) => {
      setCreatedDonation(data);
      setDonationComplete(true);
    },
  });

  const handleFileUpload = async (file) => {
    setIsUploading(true);
    try {
      const { file_url } = await base44.integrations.Core.UploadFile({ file });
      return file_url;
    } finally {
      setIsUploading(false);
    }
  };

  const handleSubmit = async () => {
    let proofUrl = null;
    if (proofFile) {
      proofUrl = await handleFileUpload(proofFile);
    }

    const donationData = {
      donor_name: isAnonymous ? 'Anonymous' : formData.donor_name,
      donor_email: formData.donor_email,
      donor_country: formData.donor_country,
      donor_country_code: formData.donor_country_code,
      amount: getFinalAmount(),
      currency: currency,
      amount_usd: getAmountInUSD(),
      donation_type: isRecurring ? 'monthly' : 'one_time',
      category: selectedCategory,
      payment_method: paymentMethod,
      bank_proof_url: proofUrl,
      status: paymentMethod === 'bank_transfer' ? 'pending' : 'pending',
      is_anonymous: isAnonymous,
      is_recurring: isRecurring,
      message: formData.message,
    };

    createDonationMutation.mutate(donationData);
  };

  if (donationComplete) {
    return (
      <div className="min-h-screen pt-20 bg-gradient-to-b from-gray-50 to-white">
        <div className="max-w-2xl mx-auto px-4 py-24 text-center">
          <motion.div
            initial={{ scale: 0 }}
            animate={{ scale: 1 }}
            className="w-24 h-24 rounded-full bg-gradient-to-r from-emerald-400 to-emerald-600 flex items-center justify-center mx-auto mb-8"
          >
            <Check className="w-12 h-12 text-white" />
          </motion.div>
          <h1 className="text-3xl md:text-4xl font-bold text-gray-900 mb-4">
            Thank You for Your Generosity!
          </h1>
          <p className="text-lg text-gray-600 mb-8">
            {paymentMethod === 'bank_transfer' 
              ? 'Your donation has been recorded. Once we verify your bank transfer, you\'ll receive a confirmation email with your receipt.'
              : 'Your donation is being processed. You\'ll receive a confirmation email shortly.'}
          </p>
          <div className="bg-white rounded-2xl p-6 shadow-lg border border-gray-100 mb-8 text-left">
            <h3 className="font-semibold text-gray-900 mb-4">Donation Summary</h3>
            <div className="space-y-3 text-gray-600">
              <div className="flex justify-between">
                <span>Amount:</span>
                <span className="font-semibold text-gray-900">
                  {selectedCurrency.symbol}{getFinalAmount().toFixed(2)} {currency}
                </span>
              </div>
              <div className="flex justify-between">
                <span>Category:</span>
                <span className="font-semibold text-gray-900">{selectedCategoryData?.title}</span>
              </div>
              <div className="flex justify-between">
                <span>Type:</span>
                <span className="font-semibold text-gray-900">
                  {isRecurring ? 'Monthly Recurring' : 'One-Time'}
                </span>
              </div>
              <div className="flex justify-between">
                <span>Reference:</span>
                <span className="font-mono text-sm text-gray-900">
                  {createdDonation?.id?.slice(0, 8).toUpperCase()}
                </span>
              </div>
            </div>
          </div>
          
          {paymentMethod === 'bank_transfer' && (
            <div className="bg-amber-50 rounded-2xl p-6 border border-amber-200 mb-8 text-left">
              <h3 className="font-semibold text-amber-800 mb-4">Bank Transfer Details</h3>
              <div className="space-y-2 text-amber-700 text-sm">
                <p><strong>Bank:</strong> Global Impact Bank</p>
                <p><strong>Account Name:</strong> Girl Child Hygiene Foundation</p>
                <p><strong>Account Number:</strong> 1234567890</p>
                <p><strong>Routing/SWIFT:</strong> GCHABCXXX</p>
                <p><strong>Reference:</strong> DON-{createdDonation?.id?.slice(0, 8).toUpperCase()}</p>
              </div>
            </div>
          )}
          
          <Button 
            onClick={() => window.location.href = '/'}
            className="bg-gradient-to-r from-rose-500 to-amber-500"
          >
            Return Home
          </Button>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen pt-20">
      {/* Hero */}
      <section className="relative py-16 overflow-hidden bg-gradient-to-br from-rose-600 via-rose-500 to-amber-500">
        <div className="absolute inset-0 opacity-10">
          <div className="absolute inset-0" style={{ backgroundImage: 'url("data:image/svg+xml,%3Csvg width=\'60\' height=\'60\' viewBox=\'0 0 60 60\' xmlns=\'http://www.w3.org/2000/svg\'%3E%3Cg fill=\'none\' fill-rule=\'evenodd\'%3E%3Cg fill=\'%23ffffff\' fill-opacity=\'0.4\'%3E%3Cpath d=\'M36 34v-4h-2v4h-4v2h4v4h2v-4h4v-2h-4zm0-30V0h-2v4h-4v2h4v4h2V6h4V4h-4zM6 34v-4H4v4H0v2h4v4h2v-4h4v-2H6zM6 4V0H4v4H0v2h4v4h2V6h4V4H6z\'/%3E%3C/g%3E%3C/g%3E%3C/svg%3E")' }} />
        </div>
        <div className="relative max-w-4xl mx-auto px-4 text-center">
          <Heart className="w-16 h-16 text-white/80 mx-auto mb-6" />
          <h1 className="text-3xl md:text-5xl font-bold text-white mb-4">
            Make a Donation
          </h1>
          <p className="text-xl text-white/90 max-w-2xl mx-auto">
            Your generosity provides dignity, education, and hope to girls who need it most.
          </p>
        </div>
      </section>

      {/* Progress Steps */}
      <div className="bg-white border-b sticky top-16 z-40">
        <div className="max-w-4xl mx-auto px-4 py-4">
          <div className="flex items-center justify-between">
            {['Choose Amount', 'Your Details', 'Payment'].map((label, index) => (
              <div key={label} className="flex items-center">
                <div className={`flex items-center justify-center w-8 h-8 rounded-full text-sm font-bold ${
                  step > index + 1 
                    ? 'bg-emerald-500 text-white' 
                    : step === index + 1 
                      ? 'bg-rose-500 text-white' 
                      : 'bg-gray-200 text-gray-500'
                }`}>
                  {step > index + 1 ? <Check className="w-4 h-4" /> : index + 1}
                </div>
                <span className={`ml-2 text-sm hidden sm:block ${step === index + 1 ? 'text-gray-900 font-medium' : 'text-gray-500'}`}>
                  {label}
                </span>
                {index < 2 && <div className="w-12 sm:w-24 h-1 bg-gray-200 mx-4" />}
              </div>
            ))}
          </div>
        </div>
      </div>

      {/* Form Container */}
      <div className="max-w-4xl mx-auto px-4 py-12">
        <AnimatePresence mode="wait">
          {/* Step 1: Choose Amount */}
          {step === 1 && (
            <motion.div
              key="step1"
              initial={{ opacity: 0, x: 20 }}
              animate={{ opacity: 1, x: 0 }}
              exit={{ opacity: 0, x: -20 }}
            >
              <h2 className="text-2xl font-bold text-gray-900 mb-8">Choose Your Donation</h2>

              {/* Category Selection */}
              <div className="grid sm:grid-cols-2 gap-4 mb-8">
                {donationCategories.map((cat) => (
                  <button
                    key={cat.id}
                    onClick={() => {
                      setSelectedCategory(cat.id);
                      setAmount(cat.amount.toString());
                      setCustomAmount('');
                    }}
                    className={`p-6 rounded-2xl border-2 text-left transition-all ${
                      selectedCategory === cat.id
                        ? 'border-rose-500 bg-rose-50'
                        : 'border-gray-200 hover:border-gray-300'
                    }`}
                  >
                    <cat.icon className={`w-8 h-8 mb-3 ${selectedCategory === cat.id ? 'text-rose-500' : 'text-gray-400'}`} />
                    <h3 className="font-bold text-gray-900 mb-1">{cat.title}</h3>
                    <p className="text-sm text-gray-600 mb-2">{cat.description}</p>
                    <div className="flex items-center justify-between">
                      <span className="font-bold text-rose-600">${cat.amount}+</span>
                      <span className="text-xs text-gray-500">{cat.impact}</span>
                    </div>
                  </button>
                ))}
              </div>

              {/* Currency Selection */}
              <div className="grid sm:grid-cols-2 gap-6 mb-8">
                <div>
                  <Label className="mb-2 block">Currency</Label>
                  <Select value={currency} onValueChange={setCurrency}>
                    <SelectTrigger>
                      <SelectValue />
                    </SelectTrigger>
                    <SelectContent>
                      {currencies.map((c) => (
                        <SelectItem key={c.code} value={c.code}>
                          {c.symbol} {c.name} ({c.code})
                        </SelectItem>
                      ))}
                    </SelectContent>
                  </Select>
                </div>
                <div>
                  <Label className="mb-2 block">Custom Amount ({selectedCurrency.symbol})</Label>
                  <Input
                    type="number"
                    placeholder="Enter custom amount"
                    value={customAmount}
                    onChange={(e) => {
                      setCustomAmount(e.target.value);
                      setAmount('');
                    }}
                    min={selectedCategoryData?.amount || 1}
                  />
                </div>
              </div>

              {/* Recurring Option */}
              <div className="bg-gray-50 rounded-2xl p-6 mb-8">
                <div className="flex items-center justify-between">
                  <div className="flex items-center gap-3">
                    <RefreshCw className="w-6 h-6 text-rose-500" />
                    <div>
                      <h3 className="font-semibold text-gray-900">Make it Monthly</h3>
                      <p className="text-sm text-gray-600">Provide consistent support with recurring donations</p>
                    </div>
                  </div>
                  <Checkbox
                    checked={isRecurring}
                    onCheckedChange={setIsRecurring}
                    className="w-6 h-6"
                  />
                </div>
              </div>

              {/* Summary */}
              <div className="bg-gradient-to-r from-rose-50 to-amber-50 rounded-2xl p-6 mb-8">
                <div className="flex items-center justify-between">
                  <span className="text-gray-700">Your Donation:</span>
                  <span className="text-3xl font-bold text-gray-900">
                    {selectedCurrency.symbol}{getFinalAmount().toFixed(2)}
                    {isRecurring && <span className="text-sm font-normal text-gray-500">/month</span>}
                  </span>
                </div>
              </div>

              <Button 
                onClick={() => setStep(2)}
                className="w-full bg-gradient-to-r from-rose-500 to-amber-500 py-6 text-lg"
                disabled={!getFinalAmount()}
              >
                Continue
                <ArrowRight className="w-5 h-5 ml-2" />
              </Button>
            </motion.div>
          )}

          {/* Step 2: Your Details */}
          {step === 2 && (
            <motion.div
              key="step2"
              initial={{ opacity: 0, x: 20 }}
              animate={{ opacity: 1, x: 0 }}
              exit={{ opacity: 0, x: -20 }}
            >
              <h2 className="text-2xl font-bold text-gray-900 mb-8">Your Details</h2>

              {/* Anonymous Option */}
              <div className="bg-gray-50 rounded-2xl p-6 mb-8">
                <div className="flex items-center justify-between">
                  <div className="flex items-center gap-3">
                    {isAnonymous ? <EyeOff className="w-6 h-6 text-gray-500" /> : <Eye className="w-6 h-6 text-rose-500" />}
                    <div>
                      <h3 className="font-semibold text-gray-900">Donate Anonymously</h3>
                      <p className="text-sm text-gray-600">Your name won't appear on the donors board</p>
                    </div>
                  </div>
                  <Checkbox
                    checked={isAnonymous}
                    onCheckedChange={setIsAnonymous}
                    className="w-6 h-6"
                  />
                </div>
              </div>

              <div className="space-y-6">
                {!isAnonymous && (
                  <div>
                    <Label className="mb-2 block">Full Name *</Label>
                    <div className="relative">
                      <User className="absolute left-3 top-1/2 -translate-y-1/2 w-5 h-5 text-gray-400" />
                      <Input
                        className="pl-10"
                        placeholder="Enter your full name"
                        value={formData.donor_name}
                        onChange={(e) => setFormData({ ...formData, donor_name: e.target.value })}
                        required
                      />
                    </div>
                  </div>
                )}

                <div>
                  <Label className="mb-2 block">Email Address *</Label>
                  <div className="relative">
                    <Mail className="absolute left-3 top-1/2 -translate-y-1/2 w-5 h-5 text-gray-400" />
                    <Input
                      type="email"
                      className="pl-10"
                      placeholder="Enter your email"
                      value={formData.donor_email}
                      onChange={(e) => setFormData({ ...formData, donor_email: e.target.value })}
                      required
                    />
                  </div>
                </div>

                <div>
                  <Label className="mb-2 block">Country *</Label>
                  <Select 
                    value={formData.donor_country_code} 
                    onValueChange={(value) => {
                      const country = countries.find(c => c.code === value);
                      setFormData({ 
                        ...formData, 
                        donor_country_code: value,
                        donor_country: country?.name || ''
                      });
                    }}
                  >
                    <SelectTrigger>
                      <SelectValue placeholder="Select your country" />
                    </SelectTrigger>
                    <SelectContent>
                      {countries.map((c) => (
                        <SelectItem key={c.code} value={c.code}>
                          {c.name}
                        </SelectItem>
                      ))}
                    </SelectContent>
                  </Select>
                </div>

                <div>
                  <Label className="mb-2 block">Message (Optional)</Label>
                  <Textarea
                    placeholder="Add a message of support..."
                    value={formData.message}
                    onChange={(e) => setFormData({ ...formData, message: e.target.value })}
                    rows={3}
                  />
                </div>
              </div>

              <div className="flex gap-4 mt-8">
                <Button 
                  variant="outline" 
                  onClick={() => setStep(1)}
                  className="flex-1"
                >
                  <ArrowLeft className="w-4 h-4 mr-2" />
                  Back
                </Button>
                <Button 
                  onClick={() => setStep(3)}
                  className="flex-1 bg-gradient-to-r from-rose-500 to-amber-500"
                  disabled={!formData.donor_email || (!isAnonymous && !formData.donor_name)}
                >
                  Continue
                  <ArrowRight className="w-4 h-4 ml-2" />
                </Button>
              </div>
            </motion.div>
          )}

          {/* Step 3: Payment */}
          {step === 3 && (
            <motion.div
              key="step3"
              initial={{ opacity: 0, x: 20 }}
              animate={{ opacity: 1, x: 0 }}
              exit={{ opacity: 0, x: -20 }}
            >
              <h2 className="text-2xl font-bold text-gray-900 mb-8">Payment Method</h2>

              {/* Payment Options */}
              <RadioGroup value={paymentMethod} onValueChange={setPaymentMethod} className="space-y-4 mb-8">
                <div className={`flex items-center gap-4 p-4 rounded-xl border-2 cursor-pointer transition-all ${
                  paymentMethod === 'bank_transfer' ? 'border-rose-500 bg-rose-50' : 'border-gray-200'
                }`} onClick={() => setPaymentMethod('bank_transfer')}>
                  <RadioGroupItem value="bank_transfer" id="bank_transfer" />
                  <Building2 className="w-6 h-6 text-gray-600" />
                  <div>
                    <Label htmlFor="bank_transfer" className="font-semibold cursor-pointer">Bank Transfer</Label>
                    <p className="text-sm text-gray-500">Transfer directly to our bank account</p>
                  </div>
                </div>
                
                <div className={`flex items-center gap-4 p-4 rounded-xl border-2 cursor-pointer transition-all opacity-50 ${
                  paymentMethod === 'card' ? 'border-rose-500 bg-rose-50' : 'border-gray-200'
                }`}>
                  <RadioGroupItem value="card" id="card" disabled />
                  <CreditCard className="w-6 h-6 text-gray-600" />
                  <div>
                    <Label htmlFor="card" className="font-semibold cursor-pointer">Card Payment</Label>
                    <p className="text-sm text-gray-500">Coming soon - Stripe, PayPal, Paystack</p>
                  </div>
                </div>
              </RadioGroup>

              {paymentMethod === 'bank_transfer' && (
                <div className="bg-amber-50 rounded-2xl p-6 mb-8 border border-amber-200">
                  <h3 className="font-semibold text-amber-800 mb-4">Bank Transfer Instructions</h3>
                  <div className="space-y-2 text-amber-700 text-sm mb-6">
                    <p><strong>Bank:</strong> Global Impact Bank</p>
                    <p><strong>Account Name:</strong> Girl Child Hygiene Foundation</p>
                    <p><strong>Account Number:</strong> 1234567890</p>
                    <p><strong>Routing/SWIFT:</strong> GCHABCXXX</p>
                  </div>
                  
                  <div>
                    <Label className="mb-2 block text-amber-800">Upload Payment Proof (Optional)</Label>
                    <div className="border-2 border-dashed border-amber-300 rounded-xl p-6 text-center bg-white">
                      <input
                        type="file"
                        accept="image/*,.pdf"
                        onChange={(e) => setProofFile(e.target.files[0])}
                        className="hidden"
                        id="proof-upload"
                      />
                      <label htmlFor="proof-upload" className="cursor-pointer">
                        <Upload className="w-8 h-8 text-amber-500 mx-auto mb-2" />
                        <p className="text-amber-700 font-medium">
                          {proofFile ? proofFile.name : 'Click to upload proof'}
                        </p>
                        <p className="text-amber-600 text-sm">PNG, JPG or PDF up to 10MB</p>
                      </label>
                    </div>
                  </div>
                </div>
              )}

              {/* Summary */}
              <div className="bg-gray-50 rounded-2xl p-6 mb-8">
                <h3 className="font-semibold text-gray-900 mb-4">Donation Summary</h3>
                <div className="space-y-3 text-gray-600">
                  <div className="flex justify-between">
                    <span>Amount:</span>
                    <span className="font-semibold text-gray-900">
                      {selectedCurrency.symbol}{getFinalAmount().toFixed(2)} {currency}
                    </span>
                  </div>
                  <div className="flex justify-between">
                    <span>Category:</span>
                    <span className="font-semibold text-gray-900">{selectedCategoryData?.title}</span>
                  </div>
                  <div className="flex justify-between">
                    <span>Type:</span>
                    <span className="font-semibold text-gray-900">
                      {isRecurring ? 'Monthly Recurring' : 'One-Time'}
                    </span>
                  </div>
                  {!isAnonymous && formData.donor_name && (
                    <div className="flex justify-between">
                      <span>Donor:</span>
                      <span className="font-semibold text-gray-900">{formData.donor_name}</span>
                    </div>
                  )}
                </div>
              </div>

              {/* Security Note */}
              <div className="flex items-center gap-3 text-sm text-gray-600 mb-8">
                <Shield className="w-5 h-5 text-emerald-500" />
                <span>Your donation is secure and encrypted. 100% goes to our programs.</span>
              </div>

              <div className="flex gap-4">
                <Button 
                  variant="outline" 
                  onClick={() => setStep(2)}
                  className="flex-1"
                >
                  <ArrowLeft className="w-4 h-4 mr-2" />
                  Back
                </Button>
                <Button 
                  onClick={handleSubmit}
                  className="flex-1 bg-gradient-to-r from-rose-500 to-amber-500 py-6"
                  disabled={createDonationMutation.isPending || isUploading}
                >
                  {createDonationMutation.isPending || isUploading ? (
                    <>
                      <Loader2 className="w-5 h-5 mr-2 animate-spin" />
                      Processing...
                    </>
                  ) : (
                    <>
                      Complete Donation
                      <Heart className="w-5 h-5 ml-2" />
                    </>
                  )}
                </Button>
              </div>
            </motion.div>
          )}
        </AnimatePresence>
      </div>
    </div>
  );
}
